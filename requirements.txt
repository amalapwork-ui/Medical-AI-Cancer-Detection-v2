# ── ML / Inference ────────────────────────────────────────────────────────────
# MUST be tensorflow-cpu, NOT tensorflow.
# The tensorflow (GPU) package loads nvcuda.dll at DLL-load time on Windows.
# This hangs indefinitely when CUDA drivers are absent or mismatched — before
# any Python code can set CUDA_VISIBLE_DEVICES to prevent it.
# tensorflow-cpu has zero CUDA code; it never scans for GPUs.
#
# Python 3.13 requires TF >= 2.19. TF 2.21 is the current stable release.
tensorflow-cpu==2.21.0

# numpy 2.1.3 is what TF 2.21 resolves at install time — do not go above 2.1.x.
# TF 2.21 is untested against numpy 2.2+.
